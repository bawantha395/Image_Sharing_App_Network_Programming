
@startuml

class User {
  -userId:string
  -password:string
  -loginStatus:string
  +verifyLogin()

}

class Admin extends User {
  -adminName:string
  -email:string
  -phoneNumber:int
  +updateCatalog()

}

class Customer extends User {
  -customerName:string
  -email:string
  -phoneNumber:int
  +signUp()
  +signIn()
  +browseCoffees()
  +setFavourite()
  +placeOrder()
  +addToCart()
  +viewOrderHistory
}

class CoffeeCatalog {
  -catalogId:string
  +addCoffeeItem()
  +removeCoffeeItem()
}

class CoffeeItem {
  -itemId: Int
  -name: String
  -specialIngredients: String
  -roastLevel: String
  -ratings: Int
  -description: String
}

class FavouriteList {
  -customerId:int
  +viewFavouriteList()
}

class ShoppingCart {
  -cartId:int
  -customerId:int
  +viewCartList()
  +updateQuantity()
  +placeOrder()

}

class Payments {
  -paymentId:int
  -paymentMethod:string
  -paymentAmount:float
  -shippingInfo:string
  +processPayment()
}

class ShippingInfo {
  -shippingId:int
  -shippingType:string
  -shippingCost:int
  -shippingRegionId:int
  +updateShippingInfo()
}
